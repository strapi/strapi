# useAppInfos

This hook is used in order to focus the user on the page in the admin panel.

## Usage

```
import { useAppInfos } from '@strapi/helper-plugin';
import { Main } from '@strapi/parts';

const HomePage = () => {
  const {
    autoReload,
    communityEdition,
    currentEnvironment,
    latestStrapiReleaseTag,
    nodeVersion,
    setUserDisplayName, // Æ’ ()
    shouldUpdateStrapi,
    strapiVersion,
    userDisplayName,
  } = useAppInfos();

  return (
    <Main>
      <h1>This is the homepage</h1>
    </Main>
  );
};
```
